PHPer的Go之路 --读写文件

package main

import (
	"fmt"
	"io/ioutil"
	"os"
)

/**
绝对路径读取文件
*/
func realRead() {
	data, err := ioutil.ReadFile("E:/goCode/test.txt")
	if err != nil {
		fmt.Println("读取文件报错：", err)
		return
	}
	fmt.Println("文件内容:", string(data))
}

/**
写入文件
 */
func writeFile() {
	// 创建文件
	f, err := os.Create("test.txt")
	if err != nil {
		fmt.Println(err)
		return
	}
	l, err := f.WriteString("哈喽鸭，德玛西亚")
	if err != nil {
		fmt.Println(err)
		f.Close()
		return
	}
	fmt.Println(l, "写入成功")
	err = f.Close()
	if err != nil {
		fmt.Println(err)
		return
	}
}

func main() {
	// 写入文件
	writeFile()
	// 读取文件内容
	realRead()
}
