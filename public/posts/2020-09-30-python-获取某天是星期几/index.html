Python 获取某天是星期几

代码

from datetime import datetime
import time

gap = 1
# 时间戳
day = time.time() - 86400*(gap-1)
# 输出年月日
ymd = time.strftime('%Y-%m-%d', time.localtime(day))
# 获取星期几 数字1-7代表周一到周日
week = datetime.strptime(ymd, '%Y-%m-%d').isoweekday()
