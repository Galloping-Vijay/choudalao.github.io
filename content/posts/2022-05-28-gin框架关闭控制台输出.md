# 代码
```go
package main

import (
	"io/ioutil"

	"github.com/gin-gonic/gin"
)

func main() {
	// 禁止Gin的控制台输出
	gin.SetMode(gin.ReleaseMode)
	gin.DefaultWriter = ioutil.Discard
	r := gin.Default()
	r.GET("/ping", func(c *gin.Context) {
		c.JSON(200, gin.H{
			"message": "pong",
		})
	})
	r.Run() // listen and serve on 0.0.0.0:8080
}
```
