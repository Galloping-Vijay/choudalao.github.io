代码如下

```php
package main

import (
	"fmt"
	"reflect"
)

type Tags struct {
	ID   int
	Name string
}

func main() {
	var tag Tags
	tag.ID = 1
	tag.Name = "魏花花"

	var typeInfo = reflect.TypeOf(tag)
	var valInfo = reflect.ValueOf(tag)
	num := typeInfo.NumField()
	for i := 0; i < num; i++ {
		key := typeInfo.Field(i).Name
		val := valInfo.Field(i).Interface()
		fmt.Println(i)
		fmt.Printf("%T --- %v\n", key, key)
		fmt.Printf("%T --- %v\n", val, val)
	}
}


```
