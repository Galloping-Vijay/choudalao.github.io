```go
package main

import (
	"fmt"
	"os"
)

var cent = `
「${blog_name}」申请友链
对方博客：${blog_name}
前往查看 ${link_url} 
`

var replaceTem = map[string]string{
	"blog_name": "潘石屹",
	"link_url":  "http://www.choudalao.com",
}

func main() {
	n := func(k string) string { return replaceTem[k] }
	s := os.Expand(cent, n)
	fmt.Println(s)
}
```
