['comments.article_id', '=', $art_id],

['comments.type', '=', 1],
['comments.pid', '=', 0],
])->select('comments.id', 'comments.pid', 'comments.content', 'comments.created_at', 'comments.user_id', 'users.name', 'users.avatar')
